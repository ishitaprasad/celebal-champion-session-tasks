select left(email, charindex('.', email) - 1) as firstname, substring(email, charindex('.', email) + 1, patindex('%[0-9]%', email) - charindex('.', email) - 1) as lastname, substring(email, patindex('%[0-9]%', email), charindex('@', email) - patindex('%[0-9]%', email)) as age, substring(email, charindex('@', email) + 1, charindex('.', email, charindex('@', email)) - charindex('@', email) - 1) as address from emails;
